<?php
// 1. 初期設定
$mode = 'input'; // 現在の状態（input:入力画面, success:完了画面）
$error_message = "";

// 2. 送信ボタンが押された時の処理
if ($_SERVER["REQUEST_METHOD"] == "POST" && isset($_POST['submit_btn'])) {
    
    // 入力データの受け取り
    $name    = h($_POST['name']    ?? '');
    $tel     = h($_POST['tel']     ?? '');
    $email   = h($_POST['email']   ?? '');
    $address = h($_POST['address'] ?? '');
    $content = h($_POST['content'] ?? '');

    // 全項目必須のチェック（サーバー側）
    if ($name && $tel && $email && $address && $content) {
        
        $to = "uchi2022.official@gmail.com";
        $subject = "【HPより】お問い合わせがありました";
        
        $body = "ホームページからのお問い合わせ内容：\n\n";
        $body .= "【お名前】: $name\n";
        $body .= "【電話番号】: $tel\n";
        $body .= "【メール】: $email\n";
        $body .= "【住所】: $address\n";
        $body .= "【内容】:\n$content\n";

        $headers = "From: " . $email;

        // メール送信実行
        if (mail($to, $subject, $body, $headers)) {
            $mode = 'success'; // 送信成功時のみ「完了画面」モードへ
        } else {
            $error_message = "サーバーのメール送信機能に問題が発生しました。";
        }
    } else {
        $error_message = "すべての項目を正しく入力してください。";
    }
}

// 安全に表示するための関数
function h($str) {
    return htmlspecialchars($str, ENT_QUOTES, 'UTF-8');
}
?>

        .success-box { background: #d4edda; color: #155724; border: 1px solid #c3e6cb; }
        .error-box { background: #f8d7da; color: #721c24; border: 1px solid #f5c6cb; }
        .form-group { margin-bottom: 20px; }
        label { display: block; margin-bottom: 8px; font-weight: bold; }
        label span { color: red; font-size: 0.8em; margin-left: 5px; }

        button { width: 100%; padding: 15px; background: #007bff; color: #fff; border: none; border-radius: 4px; font-size: 18px; cursor: pointer; font-weight: bold; }

        .back-link { display: block; text-align: center; margin-top: 20px; color: #007bff; text-decoration: none; }
    </style>
</head>
<body>

<div class="container">
    <?php if ($mode === 'success'): ?>
        <h2>送信完了</h2>
        <div class="status-msg success-box">
            送信が完了しました。<br>お問い合わせありがとうございます。
        </div>
        <a href="" class="back-link">フォームに戻る</a>

    <?php else: ?>
        <h2>お問い合わせ</h2>

        <?php if ($error_message): ?>
            <div class="status-msg error-box"><?php echo $error_message; ?></div>
        <?php endif; ?>

        <form action="" method="post">
            <div class="form-group">
                <label>お名前<span>(必須)</span></label>
                <input type="text" name="name" required placeholder="山田 太郎">
            </div>

            <div class="form-group">
                <label>電話番号<span>(必須)</span></label>
                <input type="tel" name="tel" required placeholder="09012345678">
            </div>

            <div class="form-group">
                <label>メールアドレス<span>(必須)</span></label>
                <input type="email" name="email" required placeholder="example@mail.com">
            </div>

            <div class="form-group">
                <label>住所<span>(必須)</span></label>
                <input type="text" name="address" required placeholder="東京都〇〇区1-2-3">
            </div>

            <div class="form-group">
                <label>お問い合わせ内容<span>(必須)</span></label>
                <textarea name="content" rows="6" required placeholder="こちらにご記入ください"></textarea>
            </div>

            <button type="submit" name="submit_btn">この内容で送信する</button>
        </form>
    <?php endif; ?>
